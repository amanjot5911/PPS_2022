## *Program*:  To write a code to add 2 integers
#include<stdio.h>

int main(){

    int number1, number2, sum;
    
    printf("Enter two integers: ");
    
    scanf("%d %d", &number1, &number2);
    
    sum = number1 + number2;

printf("%d + %d = %d" , number1, number2, sum);

return 0;

}

11
